int main() {
    ifstream file("demo.txt");
    if (!file.is_open()) {
        cerr << "Error opening the file!" << endl;
        return 1;
    }

    string line;
    int lineNumber = 1;
    while (getline(file, line)) {
        tokenizeLine(line, lineNumber);
        lineNumber++;
    }

    file.close();
    return 0;
}
